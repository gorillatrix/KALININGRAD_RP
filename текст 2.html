// --- STORAGE (with memory fallback) ---
function checkStorage() {
try {
localStorage.setItem('rp_test','1');
isStorageOk = localStorage.getItem('rp_test') === '1';
localStorage.removeItem('rp_test');
} catch(e) { isStorageOk = false; }
}
function safeGet(key) {
try { if(isStorageOk) return localStorage.getItem(key); } catch(e){}
if(key === 'rp_profiles') return JSON.stringify(memoryProfiles) || null;
if(key === 'rp_lastUser' && memoryProfiles._lastUser) return memoryProfiles._lastUser;
return null;
}
function safeSet(key, val) {
try { if(isStorageOk) localStorage.setItem(key, val); } catch(e){}
if(key === 'rp_profiles') {
try { memoryProfiles = JSON.parse(val); } catch(e){}
}
if(key === 'rp_lastUser') {
memoryProfiles._lastUser = val;
}
}

function isValidNick(n) {
return /^[a-zA-Z0-9]+_[a-zA-Z0-9]+$/.test(n);
}
function handleAuth(){
const nick = $('auth-nick').value.trim();
const pass = $('auth-pass').value.trim();
const pass2 = $('auth-pass2').value.trim();
const errEl = $('auth-error');
errEl.textContent = '';
if(!nick || !pass){ errEl.textContent = 'Заполни все поля!'; return; }
const isReg = $('auth-pass2').style.display !== 'none';
if(isReg){
if(!isValidNick(nick)){ errEl.textContent = 'Ник должен быть формата Nick_Name (английские буквы и _)'; return; }
if(pass !== pass2){ errEl.textContent = 'Пароли не совпадают!'; return; }
}
try {
const profiles = JSON.parse(safeGet('rp_profiles') || '{}');
if(isReg) {
if(profiles[nick]){ errEl.textContent = 'Никнейм уже занят!'; return; }
state = { nick, pass, balance:5000, level:1, xp:0, xpNext:500, satiety:100, assets:{re:[],cars:[],biz:[]}, promos:[], admin:false, isGuest:false, incomeBonus:0, timePlayed:0 };
profiles[nick] = JSON.parse(JSON.stringify(state));
safeSet('rp_profiles', JSON.stringify(profiles));
safeSet('rp_lastUser', nick);
showToast('✅ Аккаунт создан!');
} else {
if(!profiles[nick] || profiles[nick].pass !== pass){ errEl.textContent = 'Неверный ник или пароль!'; return; }
state = { ...state, ...profiles[nick] };
safeSet('rp_lastUser', nick);
showToast(`👋 Добро пожаловать, ${nick}!`);
}
startGame();
} catch(e) {
errEl.textContent = 'Ошибка: ' + e.message;
}
}

c.appendChild(d);
});

function startJob(id){
openScreen('job');
const j=JOBS.find(x=>x.id===id);
if(!j) return;
ui.jIcon.textContent=j.icon; ui.jTit.textContent=j.name; ui.jDesc.textContent=j.desc;
const pay = Math.floor(j.pay*(1+getBonus()/100));
ui.jRew.textContent=`+${pay.toLocaleString()}₽`;
state._prog = 0;
state._jobId = id;
renderTasks(j);
ui.jBar.style.width='0%'; ui.jVal.textContent='0%';
}
function renderTasks(job){
ui.jGrid.innerHTML='';
const shuffled = [...job.tasks].sort(()=>Math.random()-0.5);
shuffled.forEach(t=>{
const b = document.createElement('button');
b.className = `job-btn ${t.t==='c'?'correct':'wrong'}`;
b.innerHTML = `<span class="jb-icon">${t.i}</span><span class="jb-label">${t.l}</span><span class="jb-val">${t.v}</span>`;
b.onclick = () => {
const delta = t.t==='c' ? (12+Math.random()*3) : -(7+Math.random()*5);
jobProgress(delta);
};
ui.jGrid.appendChild(b);
});
}
function jobProgress(delta){
if(!$('screen-job').classList.contains('active')) return;
state._prog = (state._prog||0) + delta;
state._prog = Math.max(0, Math.min(100, state._prog));
ui.jBar.style.width = state._prog+'%'; ui.jVal.textContent = Math.floor(state._prog)+'%';
if(state._prog>=100) completeJob();
else {
const j = JOBS.find(x=>x.id===state._jobId);
if(j) renderTasks(j);
}
}
function completeJob(){
const j=JOBS.find(x=>x.id===state._jobId);
if(!j) return;
const pay = Math.floor(j.pay*(1+getBonus()/100));
state.balance += pay;
addXP(j.xp);
showToast(`✅ Смена завершена! +${pay.toLocaleString()}₽`);
state._prog = 0;
setTimeout(()=>startJob(state._jobId), 800);
}

});
}

updateUI();
}
